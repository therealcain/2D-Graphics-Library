# Add image manipulation:
    Color Sprite::get_pixel(const Renderer& renderer, const VectorUI& pos) noexcept
    {
        Color color;
        glReadPixels(pos.x, renderer.size.y - pos.y - 1, 1, 1, GL_RGBA, GL_FLOAT, &color);
        return color;
    }

    void Sprite::set_pixel(const Renderer& renderer, const VectorUI& pos, Color& color) noexcept
    {
        glBindTexture(GL_TEXTURE_2D, id);
        glTexSubImage2D(GL_TEXTURE_2D, 0, pos.x, pos.y, 1, 1, GL_RGBA, GL_FLOAT, &color);
        glBindTexture(GL_TEXTURE_2D, 0);
    }
